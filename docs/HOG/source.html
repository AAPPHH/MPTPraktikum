 5def calculate_gradient_directions(totalDirections = 12, offset = 0.0):
 6  angles = np.linspace(0, 2.0 * np.pi, totalDirections, endpoint=False) + offset
 7  hogDirections = np.zeros((2,totalDirections))
 8  
 9  hogDirections[0, :] = np.cos(angles)
10  hogDirections[1, :] = np.sin(angles)
11  
12  return hogDirections
13
14def hog_cell(magnitude, winningBin, totalDirections):
15  cellVector = np.zeros(totalDirections)
16  for index in range(totalDirections):
17    cellVector[index] = magnitude[winningBin == index].sum()
18    
19  return cellVector / 256.0# / np.linalg.norm(cellVector)
20
21def visualize_hog_cell(cellVector, directions, shape=(16,16)):
22  cell = np.zeros(shape)
23  W, H = shape[0], shape[1]
24  for index, value in enumerate(cellVector):
25     dx, dy = directions[0, index], directions[1, index]
26
27     x0, y0 = int(W // 2), int(H // 2)
28     x1, y1 = int(x0 + dx * W / 2), int(y0 + dy * H / 2)
29     cv2.line(cell, (x0, y0), (x1, y1), (value, value, value))
30
31  return cell
32
33def process_hog(gray, totalDirections = 12):
34  hogDirections = calculate_gradient_directions(totalDirections)
35
36  gx = cv2.Sobel(gray, cv2.CV_32F, 1, 0, ksize=5)
37  gy = cv2.Sobel(gray, cv2.CV_32F, 0, 1, ksize=5)
38  mag = np.sqrt(gx**2 + gy**2)
39
40  gx, gy = gx.flatten(), gy.flatten()
41  winningBin, bestDot = np.zeros_like(gx), np.zeros_like(gx)
42  for direction in range(totalDirections):
43    dot = gx * hogDirections[0, direction] + gy * hogDirections[1, direction]
44
45    greaterIndices = dot > bestDot
46    winningBin[greaterIndices] = direction
47    bestDot[greaterIndices] = dot[greaterIndices]
48
49  winningBin = winningBin.reshape(gray.shape) 
50
51  binGray = np.uint8(255 * winningBin / (totalDirections - 1))
52  binColor= cv2.applyColorMap(binGray, cv2.COLORMAP_JET)
53  binColor[mag < 1.5] = (0, 0, 0)
54
55  cv2.imshow("Gradient Direction", binColor)
56
57  cellSize = 8
58  hogImage = np.zeros_like(gray)
59  for x0 in range(0, gray.shape[1], cellSize):
60     for y0 in range(0, gray.shape[0], cellSize):
61        x1, y1 = x0 + cellSize, y0 + cellSize
62
63        cellVector = hog_cell(mag[y0:y1, x0:x1], winningBin[y0:y1, x0:x1], totalDirections)
64        img = visualize_hog_cell(cellVector, hogDirections, (cellSize, cellSize))
65
66        hogImage[y0:y1, x0:x1] = img
67
68  
69  cv2.imshow("HOG Vector", hogImage)
