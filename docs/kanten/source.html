 1import cv2
 2import numpy as np

 4def processImage(frame):
 5    """
 6    Process the provided image (3-channel BGR) and calculate
 7    gradients in X and Y direction as well as the gradient magnitude.
 8
 9    gx and gy shall contain the gradient direction image with values between -1 and +1
10    grad shall contain the gradient magnitude image with values between 0 and 1
11
12    :param frame: 3-channel BGR image (np.array)
13    :return: 3-tupel (gx, gy, grad) containing the gradient image in X and Y direction as well as the gradient magnitude image (1-channel np.float32 images each).
14    """
15    # TODO: Convert the image to grey using cv2.cvtColor
16    frame_gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
17
18    # TODO: Convert the image to np.float32 (divide by 255.0 first for normalization)
19    frame_gray = np.float32(frame_gray / 255.0)
20
21    # TODO: Calculate Gradients and normalize
22    gx = cv2.Sobel(frame_gray, cv2.CV_32F, 1, 0, ksize=3) / 4.0
23    gy = cv2.Sobel(frame_gray, cv2.CV_32F, 0, 1, ksize=3) / 4.0
24
25    # Calculate Gradient Magnitude and normalize
26    grad = np.sqrt(gx**2 + gy**2) / np.sqrt(2.0)
27
28    # Return 3-tupel with gradient images and gradient magnitude
29    return gx, gy, grad
