        alpha_matrix[t, :] = alpha

    # Visualize alpha vectors as heat map
    states = ["Text 1", "Text 2", "Text 3", "Text 4", "Text 5"]
    df = pd.DataFrame(
        alpha_matrix.T, index=states, columns=[f"t{t+1}" for t in range(len(sequence))]
