 30        self.X, self.P = None, None

 33    def predict(self):
 34        """
 35        Predicts the next state and covariance of the system using the Kalman filter prediction step.
 36        This method applies the state transition model to estimate the next state vector (`self.X`)
 37        and updates the state covariance matrix (`self.P`) by incorporating process noise.
 38        The prediction is based on the following:
 39          - State transition matrix (F) models the system dynamics.
 40          - Process noise covariance matrix (Q) accounts for uncertainty in the process.
 41
 42        For a 2D position and velocity model, the state vector is represented as:
 43        [x_position, y_position, x_velocity, y_velocity]. The state transition matrix F 
 44        is designed to update the position based on the current velocity, assuming a constant
 45        velocity model. The process noise covariance matrix Q is set to a small value to reflect
 46        the uncertainty in the process.  
 47
 48        Assuming constant velocity, the state transition matrix F is defined as:
 49
 50            F = [[1, 0, dt, 0],
 51                [0, 1, 0, dt],
 52                [0, 0, 1, 0],
 53                [0, 0, 0, 1]]
 54
 55        where dt is the time step between predictions. In this case, we assume dt = 1 for simplicity.
 56
 57        The process noise covariance matrix Q is defined as:  
 58
 59            Q = [[q_x, 0, 0, 0],
 60                [0, q_y, 0, 0],
 61                [0, 0, q_vx, 0],
 62                [0, 0, 0, q_vy]]
 63
 64        where q_x, q_y, q_vx, and q_vy are small values representing the process noise for position. 
 65
 66        Use np.diag to create a diagonal matrix for Q, where the diagonal elements represent the process noise variances for each state variable.^
 67        Use small noise for position and larger noise for velocity to reflect the uncertainty in the process.
 68        The prediction step updates the state vector and covariance matrix as follows:  
 69        
 70        - `X = F @ X`: This updates the state vector by applying the state transition matrix F.
 71        - `P = F @ P @ F.T + Q`: This updates the state covariance matrix by applying the state transition matrix F, its transpose, and adding the process noise covariance matrix Q.
 72        
 73        Updates:
 74          self.X: Predicted state vector after applying the state transition.
 75          self.P: Predicted state covariance matrix after accounting for process noise.
 76        """
 77        
 78        # TODO: Define the state transition matrix F
 79        F = np.array(
 80            [
 81                [1.0, 0.0, 1.0, 0.0],
 82                [0.0, 1.0, 0.0, 1.0],
 83                [0.0, 0.0, 1.0, 0.0],
 84                [0.0, 0.0, 0.0, 1.0],
 85            ]
 86        )
 87
 88        # TODO: Define the process noise covariance matrix Q
 89        Q = np.diag([0.001, 0.001, 0.1, 0.1])
 90
 91        # TODO: Predict next state using the state transition matrix
 92        self.X = F @ self.X
 93
 94        # TODO: Update the state covariance matrix using the state transition matrix and process noise
 95        self.P = F @ self.P @ F.T + Q
 96
 97    def init_filter(self, z, R):
 98        """
 99        Initializes the Kalman filter with thje first measurement and covariance.
100        
101        Parameters:
102        - z: The first measurement (2D Vector with position in X and Y).
103        - R: The covariance matrix associated with the measurement, which represents the uncertainty in the measurement.
104        
105        This method sets the initial state and covariance for the Kalman filter, which will be used
106        in subsequent prediction and update steps.
107
108        The state vector `self.X` is initialized with the first measurement, assuming a 2D position and velocity model.
109        The covariance matrix `self.P` is initialized with the measurement noise covariance `R` for position, 
110        and larger values for velocity to reflect the uncertainty in the initial state.
111        
112        Updates:
113          self.X: Predicted state vector after applying the state transition.
114          self.P: Predicted state covariance matrix after accounting for process noise.
115        """
116        # TODO: Initialize the state vector with the initial measurement, assuming a 2D position and velocity model. 
117        # Use the X and Y component of the measurement z, and set the velocity components to 0.0.
118        # HINT: The resulting state vector should be a 4D vector with the format [x_position, y_position, x_velocity, y_velocity] with shape (4, 1).
119        self.X = np.array([[z[0][0], z[1][0], 0.0, 0.0]]).T
120
121        # TODO: Initialize the covariance matrix with the measurement noise covariance R for position,
122        # and larger values for velocity to reflect the uncertainty in the initial state. 
123        # HINT: The resulting covariance matrix should be a 4x4 matrix. You can use np.diag to create a diagonal matrix.        
124        self.P = np.diag([R[0][0], R[1][1], 100.0, 100.0])
125
126
127    def update(self, z, R):
128        """
129        Performs the Kalman filter update step with the given measurement.
130        Args:
131          z (np.ndarray): The measurement vector. Shape is (2,1) for 2D position (no velocity estimates from the measurement).
132          R (np.ndarray): The measurement noise covariance matrix. Shape is (2,2) for 2D position measurements.
133        Updates:
134          self.X (np.ndarray): The state estimate after incorporating the measurement.
135          self.P (np.ndarray): The state covariance matrix after the update.
136
137        If the filter is not initialized (self.X is None), initializes the filter with the measurement.
138        Otherwise, computes the Kalman gain, updates the state estimate and covariance matrix.
139
140        First, one needs to calculate the innovation (measurement residual) `y` and the innovation covariance `S`:
141        :math:`\begin{array}{ccc}
142                y &=& z - H\cdot X\\
143                S &=& H \cdot P \cdot H^T + R
144               \end{array}`
145        
146        where `z` is the measurement, `H` is the measurement matrix, `X` is the current state estimate, and `P` is the current covariance matrix.
147
148        Then, compute the Kalman gain `K`:
149        :math:`K = P \cdot H^T \cdot S^{-1}`
150
151        Finally, update the state estimate and covariance matrix:
152
153        :math:`X = X + K  \cdot y`
154        :math:`P = (I - K \cdot H) \cdot P`
155
156        where `H` is the measurement matrix and `I` is the identity matrix
157        """                
158        H = np.array([[1.0, 0.0, 0.0, 0.0], 
159                      [0.0, 1.0, 0.0, 0.0]])
160
161        if self.X is None:
162            self.init_filter(z, R)
163            return
164        
165        y = z - H @ self.X
166        S = H @ self.P @ H.T + R
167        K = self.P @ H.T @ np.linalg.inv(S)
168
169        self.X = self.X + K @ y
170        self.P = (np.eye(4) - K @ H) @ self.P
171
172    def step(self, data):
173        # If there is a state already, predict the next state and covariance
174        if self.X is not None:
175            self.predict()
176
177            # Make a copy of the current state and covariance for visualization of the prior 
178            self.prediction = self.X.copy()
179            self.priorCov = self.P.copy()
180
181        # Update with measurements
182        for _, measurement in data["measurements"].items():
183            z, R = measurement["state"], measurement["covariance"]
184
185            if z is not None and R is not None:
186              self.update(z, R)
187
188            if self.prediction is None and self.priorCov is None:
189                self.prediction = self.X.copy()
190                self.priorCov = self.P.copy()
191
192        return {
193            "prior": {"state": self.prediction, "covariance": self.priorCov},
194            "posterior": {"state": self.X, "covariance": self.P},
195        }
