 1import cv2
 2import numpy as np
 3
 4def drawHistogram(gray):
 5    hist, _ = np.histogram(gray, bins=range(256))
 6    hist = np.float32(hist)
 7    hist = hist / np.max(hist)
 8
 9    H = 128
10    histImage = np.zeros((H, 256))
11    for bin, value in enumerate(hist):
12        x0, y0 = bin, H
13        x1, y1 = bin, int(H * (1.0 - value))
14        cv2.line(histImage, (x0, y0), (x1, y1), bin / 256.0, 1)
15
16    return histImage
17
18def adjustBrightness(gray, brightness, contrast):
19    mean, std = np.mean(gray), np.std(gray)
20
21    norm = brightness + contrast * (gray - mean) / std
22    return np.uint8(np.clip(norm, 0.0, 1.0) * 255.0)
23

47      norm = adjustBrightness(gray, brightness, contrast)

51      key = cv2.waitKey(1)

53          contrast = np.clip(contrast + 0.01, 0.0, 1.0)

57
58      if key == ord("1"):
59          brightness = np.clip(brightness - 0.05, 0.0, 1.0)
60
